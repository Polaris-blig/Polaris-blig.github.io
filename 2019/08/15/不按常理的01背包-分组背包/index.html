洛谷P1048采药 https://www.luogu.org/record/22529217

代码: 

#include<iostream>
using namespace std;

int t, m;
int tme, var;
int res[1005];

int main() {
    cin >> t >> m;
    for (int i = 0;i < m;i++) {
        cin >> tme >> var;
        for (int j = t;j >= 1;j--) {//逆序
            if (j >= tme) {
                res[j] = max(res[j], res[j - tme] + var);
            }
        }
    }
    cout << res[t];
    return 0;
}

分组背包：

洛谷P1757通天之分组背包https://www.luogu.org/problem/P1757

代码: 

#include<iostream>
#include<math.h>
using namespace std;
int n, m;
int ai, bi, ci;
int row;
int cc[105];
int cap[105][1005];
int var[105][1005];
int d[1005];

int main() {
    cin >> m >> n;
    for (int i = 1;i <= n;i++) {
        cin >> ai >> bi >> ci;
        row = max(row, ci);
        cc[ci]++;
        cap[ci][cc[ci]] = ai;
        var[ci][cc[ci]] = bi;
    }
    for (int i = 1;i <=row;i++) {
        for (int k = m;k > 0;k--) {
            for (int j = 1;j <= cc[i];j++) {
                if (k >= cap[i][j])
                    d[k] = max(d[k], d[k - cap[i][j]] + var[i][j]);
            }
        }
    }
    cout << d[m];
    return 0;
}

好吧，到这里一切正常，然后我就想用第一种搜索的方式来写一下01，结果当然是错的。
这是我的部分错误代码,vissub[]用来标记每种药材有没有被选中（原来的想法就是，从这些草药中选一部分，总重量不超过包的要求，然后来算总价值 

int t, m;
int vissum[1005];
int vissub[105];
int tme[105];
int var[105];
int d[1005];

int dfs(int s,int pos) {
    if (vissum[s]) return d[s];
    for (int i = pos;i <= m;i++) {
        if (!vissub[i] && s >= tme[i]) {
            vissub[i] = 1;
            d[s] = max(d[s], dfs(s - tme[i],i+1) + var[i]);
            vissub[i] = 0;
        }
    }
    vissum[s] = 1;
    return d[s];
}

想了很久也没想明白，后来才发现，d[s]用来记录的是，当包中剩余重量为s时，能得到的最大价值。但这么算，d[]并不能得到最大价值。
举个例子，假如有1,2,3,4,5种草药，最大重量为30，对于d[10]要选择2,4种才能得到最大价值，但是对最初dfs（30），选择了1,2,3，继续往下搜索的时候到达dfs(10),此时2以及被选择，故不能再选，所以这么搜索下去，不可能得到正确的最大价值。但是对于完全背包没有问题，因为每个都是从第一种草药开始选取。
对于前面这种想法，应该是这么写的(但是超时)： 


int maxres = 0;
void dfs(int s, int sum,int pos) {
    int ok = 0;
    for (int i = pos;i <= m;i++) {
        if (s >= tme[i]) {
            ok = 1;
            dfs(s - tme[i], sum + var[i], i + 1);
        }
    }
    if (!ok && maxres < sum)   //ok=0证明不能再往下选取
        maxres = sum;
}
int main() {
    cin >> t >> m;
    for (int i = 1;i <= m;i++) {
        cin >> tme[i] >> var[i];
    }
    dfs(t,1);
    cout << d[t];
    return 0;
}

        for (int j = 1;j <= n;j++) {
            if (j >= mon[i]) res[i][j] = max(res[i-1][j],res[i-1][j-mon[i]]+var[i]*mon[i]);

        }
    }

}

二维分组背包

洛谷P1757通天之分组背包https://www.luogu.org/problem/P1757

一开始，在网上看到的转移方程是：（d[i][j]为第i组容量为j，k为第i组第k个。
d[i][j]=max(d[i-1][j],d[i-weight[k]]+value[k])
如果只有这个方程，还不太全面，因为当假设某一组有两个商品（设重量为10，价值前者20，后者10），当要对这组计算的时候，d[i-1][j]假设已经确定为0，那么第一次d[i][j]得到20，第二次d[i][j]得到10，大值被覆盖。所以上面这个方程的毛病就是，没有与本组的其他值比较。
d[i][j]是针对于一整个组来算的，不是对于某一个物品算的。所以除了和前一组数据对比，还要和本组自己的数据对比。
所以完整的应该是： 

#include<bits/stdc++.h>
using namespace std;
struct package {
    int cap;
    int var;
}res[105][1005];
int row;
int m, n;
int ai, bi, ci;
int cnt[105];
int d[105][1005];
bool cmp(package p1, package p2) {
    return p1.cap > p2.cap;
}
int main() {
    cin >> m >> n;
    for (int i = 1;i <= n;i++) {
        cin >> ai >> bi >> ci;
        if (ai > m) continue;
        row = max(row, ci);
        cnt[ci]++;
        res[ci][cnt[ci]].cap = ai;
        res[ci][cnt[ci]].var = bi;
    }
    for (int i = 0;i <= row;i++) {
        for (int k =0 ;k<=m;k++) {
            for (int j = 1;j <= cnt[i];j++) {
                if (k >= res[i][j].cap) {
                   int r = max(d[i-1][k], d[i - 1][k - res[i][j].cap] + res[i][j].var);//暂时储存
                   d[i][k]=max(d[i][k],r);//找到最优
                }

            }
            d[i][k] = max(d[i][k],d[i - 1][k]); //不选可能更优，主要作用还是如果没有能选择的物品，那么保留上一级的值
        }
    }
    cout << d[row][m];
    return 0;
}
